====================
Simple include
====================

<% include Header %>

---

(document
  (ss_include
    (ss_delimiter_start)
    template: (ss_template_name)
    (ss_delimiter_end)))

====================
Namespaced include
====================

<% include SilverStripe/Blog/Pagination %>

---

(document
  (ss_include
    (ss_delimiter_start)

====================
Include with arguments
====================

<% include Header Title="My Page" %>

---

(document
  (ss_include
    (ss_delimiter_start)
    template: (ss_template_name)
    (ss_include_arguments
      (ss_include_argument
        name: (ss_identifier)
        value: (ss_string)))
    (ss_delimiter_end)))
